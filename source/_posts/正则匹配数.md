js正则匹配正负数，小数后只能保留两位

<!--more-->

```js
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Document</title>
</head>
<body>
    <input  onkeyup="clearNoNum(this)" class="price-input" placeholder="请输入产品溢价"/>
</body>
</html>
<script>
    function clearNoNum(obj)
        {

        //先把非数字的都替换掉，除了数字和.

        obj.value = obj.value.replace(/[^\d.\-]/g,"");


        //保证只有出现一个.而没有多个.

        obj.value = obj.value.replace(/\.{2,}/g,".");

        obj.value = obj.value.replace(/\-{2,}/g,"");

        //必须保证第一个为数字而不是.

        obj.value = obj.value.replace(/^\./g,"");

        //保证.只出现一次，而不能出现两次以上

        obj.value = obj.value.replace(".","$#$").replace(/\./g,"").replace("$#$",".");

        obj.value = obj.value.replace("-","$#$").replace(/\-/g,"").replace("$#$","-");


        //只能输入两个小数

        obj.value = obj.value.replace(/^(\-)*(\d+)\.(\d\d).*$/,'$1$2.$3');


        if(obj.value.indexOf('-') > 0) {
            obj.value = '';
        }

        }
 </script>
```
